#### 重新创建(recreate)
- 把所有旧pod全部停止, 在启动新的pod
  - 适应场景: 在测试环境的时候, 快速的把在不同节点上的pod停止, 然后在快速启动

``` yaml
[root@hdss7-200 k8s]# vim strategy-recreate.yaml 
apiVersion: apps/v1
kind: Deployment
metadata:
  name: web-recreate-22
  namespace: dev
spec:
  strategy:
    type: Recreate
  selector:
    matchLabels:
      app: web-recreate-22
  replicas: 2
  template:
    metadata:
      labels:
        app: web-recreate-22
    spec:
      containers:
      - name: web-recreate-22
        image: xusx/tomcat:8.0.22-jre8
        ports:
        - containerPort: 8080
        livenessProbe:
          tcpSocket:
            port: 8080
          initialDelaySeconds: 20
          periodSeconds: 10
          failureThreshold: 2
          successThreshold: 1
          timeoutSeconds: 5
        readinessProbe:
          httpGet:
            path: /examples/index.html
            port: 8080
            scheme: HTTP
          initialDelaySeconds: 10
          periodSeconds: 5
          failureThreshold: 1
          successThreshold: 1
          timeoutSeconds: 5
---
apiVersion: v1
kind: Service
metadata:
  name: web-recreate-22
  namespace: dev
spec:
  ports:
  - port: 80
    protocol: TCP
    targetPort: 8080
  selector:
    app: web-recreate-22
  type: ClusterIP

---
apiVersion: extensions/v1beta1
kind: Ingress
metadata:
  name: web-recreate-22
  namespace: dev
spec:
  rules:
  - host: web-recreate.chain.com
    http:
      paths:
      - path: /
        backend:
          serviceName: web-recreate-22
          servicePort: 80
```
#### 滚动更新(rollingupdate)
- `maxSurge`最大可以超出pod的百分比  
- `maxUnavailable`最大不可以25%
``` yaml
[root@hdss7-200 k8s]# vim strategy-RollingUpdate.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: web-rollingupdate
  namespace: test
spec:
  strategy:
    rollingUpdate:
      maxSurge: 25%
      maxUnavailable: 25%
    type: RollingUpdate
  selector:
    matchLabels:
      app: web-rollingupdate
  replicas: 4
  template:
    metadata:
      labels:
        app: web-rollingupdate
    spec:
      containers:
      - name: web-rollingupdate
        image: xusx/tomcat:8.0.22-jre8
        ports:
        - containerPort: 8080
        resources:
          requests:
            memory: 500Mi
            cpu: 1024m
          limits:
            memory: 500Mi
            cpu: 1024m
        livenessProbe:
          tcpSocket:
            port: 8080
          initialDelaySeconds: 20
          periodSeconds: 10
          failureThreshold: 3
          successThreshold: 1
          timeoutSeconds: 5
        readinessProbe:
          httpGet:
            path: /
            port: 8080
            scheme: HTTP
          initialDelaySeconds: 20
          periodSeconds: 10
          failureThreshold: 1
          successThreshold: 1
          timeoutSeconds: 5
---
apiVersion: v1
kind: Service
metadata:
  name: web-rollingupdate
  namespace: test
spec:
  ports:
  - port: 80
    protocol: TCP
    targetPort: 8080
  selector:
    app: web-rollingupdate
  type: ClusterIP

---
apiVersion: extensions/v1beta1
kind: Ingress
metadata:
  name: web-rollingupdate
  namespace: test
spec:
  rules:
  - host: web-rollingupdate.chain.com
    http:
      paths:
      - path: /
        backend:
          serviceName: web-rollingupdate
          servicePort: 80
```

``` shell
# 暂停升级
[root@hdss7-11 ~]# kubectl rollout pause deployment web-rollingupdate -n test 
deployment.apps/web-rollingupdate paused

# 取消暂停
[root@hdss7-11 ~]# kubectl rollout resume deployment -n test web-rollingupdate
deployment.apps/web-rollingupdate resumed

# 回滚上个版本
[root@hdss7-11 ~]# kubectl rollout undo deployment -n test web-rollingupdate
deployment.apps/web-rollingupdate rolled back
```
