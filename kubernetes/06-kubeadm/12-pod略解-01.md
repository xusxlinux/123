#### network
``` yaml
[root@hdss7-200 k8s_02]# cat pod-network.yaml 
apiVersion: v1
kind: Pod
metadata:
  name: pod-network
spec:
  containers:
  - name: web-nginx
    image: xusx/nginx:1.15.12
    ports:
    - containerPort: 80
  - name: web-tomcat
    env:
    - name: tomcat
      value: "8080"
    image: xusx/tomcat:8.0.22
    ports:
    - containerPort: 8080
      hostPort: 8080
      protocol: TCP
```

#### volume
- 一般适用于日志采集
``` yaml
[root@hdss7-200 k8s_02]# cat pod-volume.yaml 
apiVersion: v1
kind: Pod
metadata:
  name: pod-volume
spec:
  containers:
  - name: web-nginx
    image: xusx/nginx:1.15.12
    ports:
    - containerPort: 80
    volumeMounts:
    - name: shared-volume
      mountPath: /shared-web-nginx
  - name: web-tomcat
    env:
    - name: tomcat
      value: "8080"
    image: xusx/tomcat:8.0.22
    ports:
    - containerPort: 8080
      hostPort: 8080
      protocol: TCP
    volumeMounts:
    - name: shared-volume
      mountPath: /shared-web-tomcat
  volumes:
  - name: shared-volume
    hostPath:
      path: /shared-volume-data
```

- 我们要自定义hosts文件的时候可以做如下配置. 因为pod设计, 我们就要保证所有pod的hosts文件要一致
  - 配置如下参数`hostAliases`
``` yaml
[root@hdss7-200 k8s_02]# cat pod-volume.yaml 

apiVersion: v1
kind: Pod
metadata:
  name: pod-volume
spec:
  hostAliases:
  - ip: "10.59.8.44"
    hostnames:
    - "cdh4"
  containers:
  - name: web-nginx
    image: xusx/nginx:1.15.12
    ports:
    - containerPort: 80
    volumeMounts:
    - name: shared-volume
      mountPath: /shared-web-nginx
  - name: web-tomcat
    env:
    - name: tomcat
      value: "8080"
    image: xusx/tomcat:8.0.22
    ports:
    - containerPort: 8080
      hostPort: 8080
      protocol: TCP
    volumeMounts:
    - name: shared-volume
      mountPath: /shared-web-tomcat
  volumes:
  - name: shared-volume
    hostPath:
      path: /shared-volume-data
```
