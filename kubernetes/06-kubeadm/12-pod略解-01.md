#### network
``` yaml
[root@hdss7-200 k8s_02]# cat pod-network.yaml 
apiVersion: v1
kind: Pod
metadata:
  name: pod-network
spec:
  containers:
  - name: web-nginx
    image: xusx/nginx:1.15.12
    ports:
    - containerPort: 80
  - name: web-tomcat
    env:
    - name: tomcat
      value: "8080"
    image: xusx/tomcat:8.0.22
    ports:
    - containerPort: 8080
      hostPort: 8080
      protocol: TCP
```
- 得到如下测试结果
